# Adorner

装饰器,用来绘画.

创建装饰器
```c#
public class SortAdorner : Adorner
{
    //参数为被装饰的元素
    public SortAdorner(UIElement adornedElement) : base(adornedElement)
    {
        //是否阻挡鼠标
        IsHitTestVisible = false;
    }
    
    //向上的箭头
    public Geometry up= Geometry.Parse("M 0 0 L 3.5 -7 L 7 0 Z");
    
    //向下的箭头
    //public Geometry down = Geometry.Parse("M 0 0 L 3.5 7 L 7 0 Z");
        
    protected override void OnRender(DrawingContext drawingContext)
    {
        base.OnRender(drawingContext);
        
        //创建画刷
        SolidColorBrush renderBrush = new SolidColorBrush(Colors.Red);
        //创建笔
        Pen renderPen = new Pen(renderBrush, 1);
        
        //创建位置,为装饰元素的某个位子
        TranslateTransform trans = new TranslateTransform(AdornedElement.RenderSize.Width -15, AdornedElement.RenderSize.Height / 2);
        
        //应用位置
        drawingContext.PushTransform(trans);
        
        //绘制几何
        drawingContext.DrawGeometry(renderBrush, renderPen, up);        
        
        //去除上一个设置(位置)
        drawingContext.Pop();
    }
}
```
