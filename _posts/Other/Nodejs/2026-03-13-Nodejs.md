### FS模块,内置模块

开箱即用,推荐使用异步方法.

`readFile()` 

```js
var fs= require("fs");

//异步方法不会等待,所以text没有意义
//如果不写第二个参数,data为二进制,可以Tostring()
var text=fs.readFile("sample.txt","utf-8",(error,data)=>
    {
        //null
        console.log(error); 
        //hello world
        console.log(data);
    });

//undefined-------------text
console.log(text+"-------------text");

//输出顺序
//undefined-------------text
//null
//hello world
```

`readFileSync()` 同步的`readFile`

`writeFile()` 写入

```js
fs.writeFile("sample.txt",content,"utf-8",(error,data)=>
    {

    });

```

`writeFileSync()`

`rename("sample.txt","new.txt","utf-8",(error,data)=>{})` 重命名,没有同步

`unlink("sample.txt",(error,data)=>{})` 删除文件

`stat("sample.txt",(error,data)=>{})` 查看文件信息

`readdir("sample.txt",(error,data)=>{})` 读取文件夹,不会递归读取

### http模块,内置模块

```js
import http from 'http';

var server = http.createServer((req,res) => {
    //这里会输出两次,因为浏览器还会发送一个图标请求
    console.log("request received");

    //在浏览器上写入
    res.write("hello world");
    res.end();
});

server.listen(7070, () => {
    //开启服务器时执行的函数
    console.log('Server is running on port 7070');
});
```

request

`url` 属性,不带根地址

`method` 属性,请求方法(GET等待)

`headers` 属性，请求头

respond

`write()` 写入body

`end()` 响应结束

`setHeader("Content-type","text/html")` 设置响应头
 
`writeHeader()` 写入状态码

Get直接在Url里面得到.

Post用`req.on("data",(query)=>{})` query是个buffer,需要tostring.

## cookie-parser

`cookie-parser` 包 

与express关联。

服务器会发送一个cookie给浏览器,浏览器后续会自动提交.

`key`:`value` 都是字符串 key是唯一的,不能为null

`app.use(cookieparse())` 开启

`response.cookie("key","value",{options})` 发送cookie

options:
    maxAge 生存周期 60*1000 一分钟,
    httpOnly ture 不能通过文件协议(file://)  false 只能http ,普遍不开启文件协议

`response.clearCookie("key")`

`request.cookies.key` 自动读取cookie

## express-session

`express-session` 包,需要用到`cookie-parser`.

Cookie是response,Session是request.

```js
import express from 'express';
import cookieParser from 'cookie-parser';
import session from 'express-session';

var app=express();
app.use(cookieParser());

//设置加密,resave是否保存未改变的session,saveUninitialized是否保存未初始化的seesion.cookie的生命周期.
app.use(session({secret:"ha123",resave:false,saveUninitialized:false,cookie:{maxAge:100*1000}}));

app.listen(7070,()=>
    {
        console.log("server is running at http://localhost:7070")
    }
);

app.get("/",(req,res)=>
    {
        //设置session,也能直接获得
        req.session.username="ab";
        res.send("session"+req.session.username);
    }   
);

app.get("/getcookie",(req,res)=>{
    console.log(req.session.username);
    res.send("session"+req.session.username);

});

app.get("/clearcookie",(req,res)=>{
    //取消session
    req.session.destroy();
    res.send("session destroy");

});
```
